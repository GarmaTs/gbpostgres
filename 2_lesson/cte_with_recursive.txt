WITH RECURSIVE dept_chain (id, parent_id, name) AS (
    SELECT id, parent_id, name
    FROM departments
    WHERE name = 'Romulus'
    UNION
    SELECT d.id, d.parent_id, d.name
    FROM departments d
    JOIN dept_chain dc
        ON d.id = dc.parent_id
)
select * from dept_chain;
